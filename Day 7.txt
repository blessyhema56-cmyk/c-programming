#include <stdio.h>
#include <string.h>

struct Address {
    char city[100];
    int pincode;
};

struct Student {
    int id;
    char name[100];
    struct Address address;
};

void printStudentDetails(struct Student *student) {
    printf("Student ID: %d\n", student->id);
    printf("Student Name: %s\n", student->name);
    printf("City: %s\n", student->address.city);
    printf("Pincode: %d\n", student->address.pincode);
}

int main() {
    struct Student student;
    student.id = 1;
    strcpy(student.name, "John Doe");
    strcpy(student.address.city, "New York");
    student.address.pincode = 10001;

    printf("Student Details:\n");
    printStudentDetails(&student);

    return 0;
}
Output
Student Details:
Student ID: 1
Student Name: John Doe
City: New York
Pincode: 10001

#include <stdio.h>
#include <string.h>
#include <stdlib.h>

struct Supplier {
    char name[100];
    char location[100];
};

struct Product {
    int id;
    char name[100];
    float price;
    struct Supplier supplier;
};

void printProductDetails(struct Product *product) {
    printf("Product ID: %d\n", product->id);
    printf("Product Name: %s\n", product->name);
    printf("Product Price: %.2f\n", product->price);
    printf("Supplier Name: %s\n", product->supplier.name);
    printf("Supplier Location: %s\n", product->supplier.location);
}

int main() {
    struct Product *product = (struct Product *)malloc(sizeof(struct Product));
    product->id = 1;
    strcpy(product->name, "Laptop");
    product->price = 999.99;
    strcpy(product->supplier.name, "ABC Inc.");
    strcpy(product->supplier.location, "New York");

    printf("Product Details:\n");
    printProductDetails(product);

    free(product);

    return 0;
}
Output
Product Details:
Product ID: 1
Product Name: Laptop
Product Price: 999.99
Supplier Name: ABC Inc.
Supplier Location: New York

#include<stdio.h>
#include<stdlib.h>
struct contact{
    long int number;
    struct contact *next;
};
int main(){
    //storage allocation
    struct contact *ph1 = (struct contact*)malloc(sizeof(struct contact));
    struct contact *ph2 = (struct contact*)malloc(sizeof(struct contact));
    struct contact *ph3 = (struct contact*)malloc(sizeof(struct contact));
    //storing numbers
    ph1->number = 9487606290;
    ph2->number = 9500556890;
    ph3->number = 9659626251;
    //interconnection
    ph1->next = ph2;
    ph2->next = ph3;
    ph3->next = NULL;
    //pointing head
    struct contact *head = ph1;
    //display
    printf("Contact:\n");
    while(head != NULL){
        printf("%ld ---->",head->number);
        head = head->next;
    }
    printf("finish");
    return 0;
}
OUTPUT

9487606290 ---->9500556890 ---->9659626251 ---->finish
